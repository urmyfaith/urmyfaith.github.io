elf('#box').html('Hello world!').css({
	textAlign: 'center'
}).tween({
	property: {fontSize: {to: 200, unit: '%'}}
}).on('click', function (ev) {
	elf('#box').tween({
		property: {width: {from: 100, to: 40, unit: '%'}},
		oncomplete: function () {
			elf('#source').css({
				display: 'block',
				opacity: 0
			}).tween({
				property: {
					opacity: {to: 1},
					width: {to: 60, unit: '%'}
				}
			});
		}
	});
});
